#include<iostream>
using namespace std;
int main (){
    int n,ar[100000],count=0,max;
    cin>>n;
    max=0;
    for (int i=0; i<n; i++){
        cin>>ar[i];
        if (ar[i]>max){
            max=ar[i];
        }
    }
    for (int i=0; i<n; i++){
        if (ar[i]==max){
            count++;
        }
    }
    cout<<count;
}
